\subsection{ConvertIntToBytes}
\label{func:ConvertIntToBytes}
\begin{pdfunction}
{ConvertIntToBytes(U32\_t integer, U8\_t num\_bytes,  U8\_t out\_bytes) }
{ 
Convert an integer value into separate bytes. This allows 
easy splitting of for instance 32-bits value into 4 bytes. 
bytes will be stored with the LSB at 0. }
\pdfunctionargin{(U32_t) integer}{ Integer value. }
\pdfunctionargin{(U8_t)num_bytes}{ Number of bytes present in the integer value. 1-4. }
\pdfunctionreturn{(U8_t) }{result. }{0}{Number of bytes has an illegal value i.e. 0 or >4. }
\pdfunctionreturn{(U8_t) }{result. }{1}{Operation successful. }
\end{pdfunction}
